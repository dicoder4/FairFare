####open first command prompt - Discovery Service
cd discovery-service
.\mvnw spring-boot:run
# Wait for it to start, then check: http://localhost:3000

####open second command prompt - User Service  
cd user-service
.\mvnw spring-boot:run
# Test endpoints: http://localhost:3001/users

####open third command prompt - Billing Service
cd billing-service
.\mvnw spring-boot:run
# Test endpoints: http://localhost:3003/bills

####open fourth command prompt - Splitter Service
cd splitter-service
.\mvnw spring-boot:run
# Test endpoints: http://localhost:3002/splits

####open fifth command prompt - Gateway Service
.\mvnw spring-boot:run
# Gateway routes: 
# http://localhost:3081/users
# http://localhost:3081/groups  
# http://localhost:3081/bills
# http://localhost:3081/splits
